◆ vue3 공식 홈페이지 라이프사이클 훅
https://vuejs.org/guide/essentials/lifecycle.html#lifecycle-diagram

◆ 컴포넌트 라이프사이클
: 컴포넌트의 라이프 사이클은 크게 4단계로 구분된다.
Creation(생성) -> Mounting(장착) -> Updating(변경) -> Destruction(소멸)

▶ Creation(생성)
: 컴포넌트 초기화 단계이며, 컴포넌트가 DOM에 추가되기 전이므로 DOM에 접근 불가
setup() -> beforeCreate() -> created()
----------------------------------------------------------------------------------------
export default {
  beforeCreate() {
// 컴포넌트 인스턴스가 초기화될때 실행(OptionsAPI)
  },
  created() {
// 인스턴스 초기화 완료 후 호출되는 훅(OptionsAPI)
  },
  setup() {
// beforeCreate/created는 vue3(CompositionAPI)에서는  setup()에서 대체할 수 있다.
  },
}
----------------------------------------------------------------------------------------

▶ Mounting(장착)

- 1. onBeforeMount()
: 컴포넌트가 마운트 되기전에 호출 (사용 권장X)

- 2. onMounted()
: 컴포넌트가 마운트 된 후에 호출되기 때문에 DOM에 접근이 가능하다.
=> 자식 컴포넌트의 onMounted가 모두 완료된 이후에 호출

----------------------------------------------------------------------------------------
export default {
  setup() {
	onBeforeMount(()=> {
		console.log("마운트전");
	});

	onMounted(()=> {
		console.log("마운트완료");
	});
  },
}
----------------------------------------------------------------------------------------

▶ Updating(변경)
: DOM안에 반응형 데이터가 변경되거나, 어떠한 이유로 재렌더링이 발생할 때 호출
=> 디버깅 시 재렌더링 시점을 알아볼때도 유용하게 쓰일 수 있다

- 1. onBeforeUpdate()
: DOM트리를 업데이트 하기 직전에 호출

- 2. onUpdated()
: DOM트리 업데이트가 완료된 후에 호출

----------------------------------------------------------------------------------------
export default {
  setup() {
	onBeforeUpdate(()=> {
		console.log("업데이트 되기전 DOM상태를 가지고 있음");
	});

	onUpdated(()=> {
		console.log("업데이트 된 DOM상태를 가지고 있음");
	});
  },
}
----------------------------------------------------------------------------------------

▶ Destruction(소멸)
: 컴포넌트 v-if로 테스트 가능

----------------------------------------------------------------------------------------
export default {
  setup() {
	onBeforeUpdate(()=> {
		console.log("언마운트 되기전 DOM상태를 가지고 있음");
	});

	onUpdated(()=> {
		console.log("언마운트 된 이후 DOM상태는 없음");
	});
  },
}
----------------------------------------------------------------------------------------
